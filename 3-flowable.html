Script Task Rev. Cupos

Revisar cupos disponibles

Groovy Rev. Cupos

import groovy.json.JsonSlurper

def cursoBuscado = execution.getVariable("curso")
def url          = "http://localhost:8080/gestion-cursos/api/curso"
def json         = url.toURL().text
def slurper      = new JsonSlurper()
def cursos       = slurper.parseText( json )

for(curso in cursos) {
    if(curso.nombre == cursoBuscado) {
        execution.setVariable("cupos", curso.cuposDisponibles)
    }
}

Condiciones Gateways (UEL)

<conditionExpression xsi:type="tFormalExpression"><![CDATA[${vars:greaterThan(cupos, 0)}]]></conditionExpression>

Groovy Inscripción

import groovy.json.JsonOutput
import groovy.json.JsonSlurper

def nombre        = execution.getVariable("nombre")
def email         = execution.getVariable("email")
def curso         = execution.getVariable("curso")
def url           = "http://localhost:8080/gestion-cursos/api/curso/inscribir/"+curso
def conexion      = url.toURL().openConnection() as HttpURLConnection

conexion.doOutput = true
conexion.requestMethod = "POST"
conexion.setRequestProperty("Content-Type", "application/json")
conexion.setRequestProperty("Accept", "application/json")
def payload = [
    nombre: nombre,
    email: email
]
def jsonPayload = JsonOutput.toJson(payload)

conexion.outputStream.withWriter { writer ->
    writer << jsonPayload
}

def responseCode = conexion.responseCode
// 400 error // 200 ok
if( responseCode == 200 ) {
    println(responseCode)
    println(conexion.content.text)
} else {
    println(responseCode)
    println("Ha ocurrido un error")
}

Java Delegates

  • Clases Java que se pueden vincular al modelo BPMN

  • Implementan la interfaz org.flowable.engine.delegate.JavaDelegate

  • Se debe generar JAR como librería y cargar archivo en flowable-rest/WEB-INF/lib

Código Notificación Inscripción

public class EnviarNotificacionInscripcion implements JavaDelegate {

    @Override
    public void execute(DelegateExecution execution) {
        NotificadorImpService service = new NotificadorImpService();
        NotificadorImp port = service.getNotificadorImpPort();

        String nombre   = execution.getVariable("nombre", String.class);
        String email    = execution.getVariable("email", String.class);
        String curso    = execution.getVariable("curso", String.class);

        String respuesta = port.notificarInscripcion(nombre, email, curso);
        System.out.println( respuesta );
    }
}
